#version 330 core
out vec4 FragColour;
in vec3 drawColour;
in vec2 TextureCoords;

uniform sampler2D containerTex;
uniform sampler2D awesomefaceTex;
uniform float textureMix;
void main() {
  FragColour = mix(texture(containerTex, TextureCoords), 
                   texture(awesomefaceTex, TextureCoords), textureMix);
}
