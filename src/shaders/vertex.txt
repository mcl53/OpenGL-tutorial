#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in vec3 aColour;
layout (location = 2) in vec2 aTextureCoords;
out vec3 drawColour;
out vec2 TextureCoords;

void main() {
  gl_Position = vec4(aPos.x, aPos.y, aPos.z, 1.0f);
  drawColour = aColour;
  TextureCoords = aTextureCoords;
}
